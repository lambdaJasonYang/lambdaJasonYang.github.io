#assume r is fac(n-1)
#design facInc to equal fac(n)
#facInc = n * fac(n-1)
def facInc(n,r):
  return n*r

#basecase is i == 0 then return r = 1
#facInc(i,r) is i * fac(i-1)
#when i == n, it becomes fac(n,r) = n * fac(n-1)
def fac3(n):
  i = 0
  r = 1
  while(i != n):
    i = i + 1
    r = facInc(i,r)
  return r

#due to associativity
def fac4(n):
  i = 0
  r = 1
  x = n
  while(i != x):
    r = facInc(r,x)
    x = x-1
  return r
