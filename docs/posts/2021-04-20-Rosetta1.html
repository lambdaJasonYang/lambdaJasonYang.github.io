interface CarInterface {
    model: string;
    year: number;
}

let somecar: CarInterface = {
    model : "Honda",
    year : 1996
}
//Class
class electricCar implements CarInterface {
    model : string;
    year : number;
    private msg : string;
    constructor(x: string, y : number, z : string){
        this.model = x;
        this.year = y; 
        this.msg = z;
    }
    outp = ():string => {
        return this.msg;
    }
}

let tesla = new electricCar("model E",2012,"Honk Honk")
console.log(tesla.outp());
