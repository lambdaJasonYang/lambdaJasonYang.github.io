    golang fmt IGNORES WHITESPACES ” bleh” => “bleh”

    package main
import (
	"fmt"
)
func main() {
	var p int
	fmt.Scanf("%d", &p)
	fmt.Println(p)
}
    	var v1,v2 int64

    Golang automatically strips out white-space for inputs
    To stop that, use bufio

    import(
  "bufio"
  "os"
)

v1 := bufio.NewReader(os.Stdin)
line, err := v1.ReadString('\n')
