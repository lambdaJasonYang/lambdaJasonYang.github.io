    Calling Github APIs

    Posted on August 10, 2021
    Tags: prog, cloud, frontend

    Get your repo specific Personal Access Token(PAT) at https://github.com/settings/personal-access-tokens

    https://api.github.com/repos/userJY/{YOUR REPO HERE}/contents/{PATH TO FILE HERE}

    const axios = require('axios');

const a = async () => {
const response = await axios.get('https://api.github.com/repos/UserJY/myRepoName', {
    headers: {
        'Authorization': 'token github_pat_XXXXXXXXXXXXXXXXXXXXXXXXXXXXXX',
        'Accept': 'application/vnd.github.v3.raw',
        "Content-Type": "application/json"
    }
});

  console.log(response)
}
